  3. 4lex

    Thanks it worked I just made these which I call on to check what it is. Strangely the stag is Horse as well
    Code:
    {% raw %}
    private bool IsAnimal(BaseEntity entity)
        {
            if (entity.GetType().ToString() == "Boar" ||
                entity.GetType().ToString() ==  "Horse" ||
                entity.GetType().ToString() ==  "Bear" ||
                entity.GetType().ToString() ==  "Wolf")
                return true;
            else return false;
        }        private bool IsBuilding(BaseEntity entity)
        {
            if (entity.GetType().ToString() == "SimpleBuildingBock" ||
                entity.GetType().ToString() == "BuildingBlock" ||
                entity.GetType().ToString() == "BuildingPrivledge" ||
                entity.GetType().ToString() == "Door" ||
                entity.GetType().ToString() == "Baricade")
                return true;
            else return false;
        }        private bool IsWorldEntity(BaseEntity entity)
        {
            if (entity.GetType().ToString() == "BaseCorpse" ||
                entity.GetType().ToString() == "LootContainer")
                return true;
            else return false;
        }
    {% endraw %}
