  1. Kunazai

    @k1lly0u did some tracing in the mod to figure out what was throwing a null reference exception. Looks like player.GetComponent<GunGamePlayer>() is throwing the exception in the timer of GiveGear(). Not sure if this is an oxide issue, but thought I'd point it out to help you start looking in the right direction.
    Code:
    {% raw %}
     private void GiveGear(BasePlayer player)
        {
            if (gameEnding) return;
            player.inventory.Strip();
            timer.In(1, () =>
            {
                GiveRankKit(player, player.GetComponent<GunGamePlayer>().level);
                if (!downgradeDisabled)
                {
                    GiveItem(player, configData.GameSettings.DowngradeWeapon);
                }
                foreach (var entry in weaponSet.PlayerGear)
                {
                    GiveItem(player, entry);
                }
            });
        }
    {% endraw %}
