           void OnEntityDeath(BaseEntity entity, HitInfo info)

           {
                BasePlayer Killed = entity.ToPlayer();
                KDDict[Killed.userID].Deaths= KDDict[Killed.userID].Deaths+1;
                GUI(Killed, Update:true, Killed:true);
