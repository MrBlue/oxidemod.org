  3. DylanSMR

    Example:
    Code:
    {% raw %}
    var foundentities = UnityEngine.Object.FindObjectsOfType<BaseEntity>();
var  prefabID = 1234;
foreach(var entity in  foundentities)
{
    if(entity.prefabID = prefabID)
    {
         entity.Kill();
    }
}
    {% endraw %}
    This checks every entity in the server for(prefabID) and if it matches it kills the entity.

    Example for turrets is the same thing. Just do:
    Code:
    {% raw %}
    var foundent = UnityEngine.Object.FindObjectsOfType<BaseEntity>(); 
foreach(var turret in foundent)
{
    if(turret.OwnerID == corpseOwnerID)
    {
        turret.Kill();
    }
}
    {% endraw %}
