  2. Norn

    Try this:
    Code:
    {% raw %}
    var unix = Math.round(+new Date()/1000);
var date = new Date(unix * 1000);
var currentDate = ('0' + date.getDate()).slice(-2) + '/' + ('0' + (date.getMonth() + 1)).slice(-2) + '/' + date.getFullYear() + ' ' + ('0' + date.getHours()).slice(-2) + ':' + ('0' + date.getMinutes()).slice(-2);
    {% endraw %}
