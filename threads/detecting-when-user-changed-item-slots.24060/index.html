  2. Mr. Blue

    You could use the OnPlayerInput here...

    Code:
    {% raw %}
    void OnPlayerInput(PlayerSession session, InputControls input)
{
        if (input.Hotbar1)
        {
               hurt.SendChatMessage(session, "Hotbar1");
         }
}
    {% endraw %}
