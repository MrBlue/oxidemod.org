  2. FuJiCuRa

    Pay attention to this scheme, then you might know how you technically could manage it (with care):
    Code:
    {% raw %}
    // Network.Server
public Networkable CreateNetworkable(uint uid)
{
    Networkable networkable = Pool.Get<Networkable>();
    networkable.ID = uid;
    networkable.sv = this;
    if (uid > this.lastValueGiven)
    {
        this.lastValueGiven = uid;
    }
    return networkable;
}
    {% endraw %}
