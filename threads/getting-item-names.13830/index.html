  3. Tuntenfisch

    Example console command:
    Code:
    {% raw %}
    [ConsoleCommand("items.dump")]
void Dump(ConsoleSystem.Arg arg)
{
    Dictionary<string, string> items = new Dictionary<string, string>();    foreach (ItemDefinition definition in ItemManager.itemList)
    {
        items.Add(definition.displayName.english, definition.shortname);
    }    Interface.Oxide.DataFileSystem.WriteObject("items.dump", items);
}
    {% endraw %}
