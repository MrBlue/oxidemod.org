    This should work:
    Code:
    {% raw %}
    int GetContentsInMililitres(Item item)
{
    if (item.contents == null || item.contents.allowedContents != ItemContainer.ContentsType.Liquid) return -1;    int amount = 0;
    foreach (Item i in item.contents.itemList)
    {
        amount += i.amount;
    }
    return amount;
}
    {% endraw %}
