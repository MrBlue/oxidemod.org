    Code:
    {% raw %}
    local function split(str, sep)
    local tbl = {}
    for word in string.gmatch(str, "[^"..sep.."]+") do
        table.insert(tbl, word)
    end
    return tbl
endlocal string = "this is:your string"
local splitted = split(string, ":")-- output
for key, value in pairs(splitted) do
    print(key, value)
end
