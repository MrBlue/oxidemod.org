  2. DylanSMR

    I would just create a text feature right next to it,
    Code:
    {% raw %}
                elements.Add(new CuiLabel
            {
                Text =
                {
                    Text = "Insert Text Here",
                    FontSize = 16,
                    Align = TextAnchor.MiddleCenter,
                    Color = "255 255 0"
                },
                RectTransform =
                {
                    AnchorMin = $"0.00 0.00",
                    AnchorMax = $"0.00 0.00"
                }
            }, GUIBackground);
    {% endraw %}
