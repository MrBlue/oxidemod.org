  4. EnigmaticDragon

    Actually DroppedItem has the SetParent(BaseEntity) method, since it is derived from BaseEntity. So this should work:

    Code:
    {% raw %}
    BaseEntity box; // your box
DroppedItem worldModel; // the laptop
[...]
worldModel.SetParent(box);
    {% endraw %}
    I've tested it with this code, works for me:
    Code:
    {% raw %}
    [ChatCommand("laptop")]
void Laptop(BasePlayer player, string command, string[] args)
{
    RaycastHit hitinfo;
    if (Physics.Raycast(player.eyes.position, Quaternion.Euler(player.GetNetworkRotation()) * Vector3.forward, out hitinfo, 30f,
        LayerMask.GetMask(new string[] { "Construction", "Terrain" })))
    {
        BaseEntity box = GameManager.server.CreateEntity(
            "assets/prefabs/deployable/large wood storage/box.wooden.large.prefab",
            hitinfo.point,
            Quaternion.Euler(0, player.GetNetworkRotation().y + 180, 0));
        box.Spawn();        DroppedItem worldModel = ItemManager.CreateByItemID(1490499512).Drop(new Vector3(0, 0.75f, 0), Vector3.zero, Quaternion.Euler(0, 180, 0)).GetComponent<DroppedItem>();        worldModel.SetParent(box);        // if you don't want the laptop to move
        worldModel.GetComponent<Rigidbody>().isKinematic = true;
        worldModel.GetComponent<Rigidbody>().useGravity = false;        // if you don't want players to pick it up
        worldModel.allowPickup = false;
        // no despawn
        worldModel.Invoke("IdleDestroy", float.MaxValue);
    }
}
    {% endraw %}
    Edit 1: I just noticed that "worldModel.Invoke("IdleDestroy", float.MaxValue)" does not prevent the item from despawning... "worldModel.CancelInvoke("IdleDestroy")" doesn't seem to work either.

    Edit 2: Okay, this should work to prevent despawning:
    Code:
    {% raw %}
    worldModel.CancelInvoke((Action)Delegate.CreateDelegate(typeof(Action), worldModel, "IdleDestroy"));
    {% endraw %}

  9. EnigmaticDragon

    Since "OnParentRemoved()" (which also resets the despawn timer) gets called for the laptop when the box is destroyed, the laptop would get despawned after some time anyway. However, if you want to destroy it immediately, try this:
    Code:
    {% raw %}
    void OnEntityDeath(BaseCombatEntity entity, HitInfo info)
{
    if (entity.ShortPrefabName.Equals("box.wooden.large")) // "woodbox_deployed" for small box
        if (entity.children?[0].ShortPrefabName.Equals("generic_world") ?? false)
            entity.children[0].Kill();
}
    {% endraw %}
