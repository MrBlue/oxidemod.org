  8. Calytic Community Admin Community Mod

    Code:
    {% raw %}
    using System;
using Oxide.Core.Plugins;
using Rust;
using UnityEngine;namespace Oxide.Plugins
{
    [Info("ShutdownCancelCraft", "Calytic", "0.0.1")]
    [Description("Cancels crafting when restarting")]
    class ShutdownCancelCraft : RustPlugin
    {
        [ConsoleCommand("global.restart")]
        private void ccGlobalRestart(ConsoleSystem.Arg arg)
        {
            rust.RunServerCommand("restart");
            timer.In(50, delegate()
            {
                foreach (BasePlayer player in BasePlayer.activePlayerList)
                {
                    player.inventory.crafting.CancelAll(true);
                }
            });
        }
    }
}
    {% endraw %}
