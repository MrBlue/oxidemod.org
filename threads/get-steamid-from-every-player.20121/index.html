                string permisos = string.Join(", ", permission.GetUserPermissions(entry.Id.ToString()));
                if (String.IsNullOrEmpty(permisos)) continue;
                string usuario = entry.Name.Replace("'", "");
                Usado(usuario, entry.Id, permisos);
            }
