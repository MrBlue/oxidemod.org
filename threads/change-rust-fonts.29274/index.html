  3. HOUGAN_Y

    If you want to change it in GUI, use something like:
    Code:
    {% raw %}
    container.Add(new CuiButton
{
    RectTransform = { AnchorMin = "0.7470833 0.1138898", AnchorMax = "0.8546874 0.1435185" },
    Button = { Color = "0 0 0 0.6" },
    Text = { Text = $"H<color=#DC143C>Ø</color>UGAN PR<color=#DC143C>Ø</color>JECT", Align = TextAnchor.MiddleCenter, Font = "robotocondensed-regular.ttf", FontSize = 14 }
}, "XP", "PROJECT");
    {% endraw %}
