    Hey kamil make sure you have the commands enabled in your default group like so:
    Code:
    {% raw %}
    groups {
    default {
        default = 'True'
        inherits = 'guest'
        nameFormat = '%name%'
        chatFormat = '%name% : %message%'
        guildFormat = '[00CC00][%guild%] %chatFormat%[-]'
        permissions {
            - 'rok.fairplay'
            - 'rok.command.help.list'
            - 'rok.health.suicide'
        }
    }
    {% endraw %}
