    No i didn't, but made a quick test case for that thing.
    It were working, however it is untested for now. Here you go:
    Code:
    {% raw %}
    void OnEntityEnter(TriggerBase trigger, BaseEntity entity)
{
    if (entity.ToPlayer())
    {
        var plr = entity.ToPlayer();
        var beartrap = trigger.GetComponent<BearTrapTrigger>();
        if (beartrap)
            timer.In(0.8f, () => { if (!beartrap._trap.Armed()) beartrap._trap.Arm(); });
    }
}
