    Code:
    {% raw %}
    // BasePlayer
public Item GetActiveItem()
{
    if (this.svActiveItemID == 0u)
    {
        return null;
    }
    if (this.IsDead())
    {
        return null;
    }
    return this.inventory.containerBelt.FindItemByUID(this.svActiveItemID);
}// Item
public int position;
