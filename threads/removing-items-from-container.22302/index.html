  3. FuJiCuRa

    Code:
    {% raw %}
            void ClearContainer(ItemContainer container)
        {
            while (container.itemList.Count > 0)
            {
                var item = container.itemList[0];
                item.RemoveFromContainer();
                item.Remove(0f);
            }
        }
    {% endraw %}
