  2. Wulf Community Admin

    Pretty sure you'd have to loop through all the players and grab their bag locations.

    Here's some code I use in my Respawner plugin:
    Code:
    {% raw %}
    function PLUGIN:FindSleepingBagByPlayer(player)
    local bags = {}
    local sleepingBags = UnityEngine.Object.FindObjectsOfTypeAll(global.SleepingBag._type)
    if sleepingBags.Length == 0 then return false end
    for i = 0, tonumber(sleepingBags.Length - 1) do
        local sleepingBag = sleepingBags[i];
        if sleepingBag.deployerUserID and sleepingBag.deployerUserID == player.userID then bags[#bags + 1] = sleepingBag end
    end
    if #bags == 0 then return false end
    return bags
end
    {% endraw %}

  12. Mughisi

    Why don't you use reflection?

    Code:
    {% raw %}
    local FindForPlayer = global.SleepingBag._type:GetMethod("FindForPlayer")
function PLUGIN:FindSleepingBags(steamID)
    param = util.TableToArray( { steamID } )
    util.ConvertAndSetOnArray( param, 0, steamID, System.UInt64._type )    return FindForPlayer:Invoke( nil, param )
end
    {% endraw %}
    [DOUBLEPOST=1421829061][/DOUBLEPOST]Example:
    Code:
    {% raw %}
    function PLUGIN:cmdTest( player, cmd, args )
    steamID = rust.UserIDFromPlayer( player )

end
