  5. Cruser

    These work for me. Note that i dont have them setup as mods but as admins, but they do have the rights you ask for:

    Code:
    {% raw %}
    # guest {   # Unregistered users will use these permissions.
# nameFormat = '%name%'
# chatFormat = '%name% : %message%'
# guildFormat = '[00CC00][%guild%] %chatFormat%[-]'
# }
groups {
    default {
        default = 'True'
        nameFormat = '%name%'
        chatFormat = '[FFFFFF]%name% [000000]: [FFFFFF]%message%'
        guildFormat = '[67fcd2][%guild%] %chatFormat%[-]'
        permissions {
            - 'rok.chat.me'
            - 'rok.chat.guild'
            - 'rok.health.suicide'
            - 'rok.help.list'
            - 'rok.fairplay'
        }
        inherits = ''
    }
    vip {
        nameFormat = '%name%'
        chatFormat = '[FFFFFF][[00FFFF]VIP[FFFFFF]] [FF6600]%name% [000000]: [00FFFF]%message%'
        guildFormat = '[67fcd2][%guild%] %chatFormat%[-]'
        permissions {
            - 'rok.chat.me'
            - 'rok.chat.guild'
            - 'rok.health.suicide'
            - 'rok.help.list'
            - 'rok.fairplay'
        }
        inherits = 'default'
    }
    mod {
        nameFormat = '%name%'
        chatFormat = '[10A1FF]MOD[FFFFFF]%name% [000000]: [FFFFFF]%message%'
        guildFormat = '[67fcd2][%guild%] %chatFormat%[-]'
        permissions {        }
        inherits = ''
    }
    admin {
        nameFormat = '%name%'
        chatFormat = '[FFFFFF]%name% [000000]: [FFFFFF]%message%'
        guildFormat = '[67fcd2][%guild%] %chatFormat%[-]'
        permissions {
            - 'codehatch.admin.kick'
            - 'codehatch.admin.ban'
            - 'rok.command.admin.godmode'
            - 'rok.command.teleport'
            - 'rok.command.teleport.coord'
            - 'rok.command.teleport.user'
            - 'codehatch.command.admin.notice'
            - 'rok.command.ui.nametag.show'
            - 'rok.command.ui.nametag.hide'
            - 'rok.chat.me'
            - 'rok.chat.guild'
            - 'rok.health.suicide'
            - 'rok.help.list'
            - 'rok.fairplay'
        }
        inherits = ''
    }
    owner {
        nameFormat = '%name%'
        chatFormat = '[FFFFFF]%name% [000000]: [FFFFFF]%message%'
        guildFormat = '[67fcd2][%guild%] %chatFormat%[-]'
        permissions {
            - '*'
        }
        inherits = ''
    }
}
users {
    Gypsy {
        groups {
            - 'Owner'
            - 'default'
        }
        permissions {
        }
    }
    Creeper {
        groups {
            - 'Owner'
            - 'default'
        }
        permissions {
        }
    }
    Alex {
        groups {
            - 'admin'
            - 'default'
        }
        permissions {
        }
    }
    Webs2020 {
        groups {
            - 'admin'
            - 'default'
        }
        permissions {
        }
    }
}macros {
}
