    Code:
    {% raw %}
                    foreach (BaseEntity entity in BaseNetworkable.serverEntities.Where(p => (p as BaseEntity).net.ID == StoredNetID).ToList())
                {
                    entity.Kill();
                }
    {% endraw %}
