    //For Group
    if ( permission.GroupHasPermission("GROUPNAME", "MyPlugin.MyPermission") )
    {
    // do stuff
    }

    // For User
    if ( permission.UserHasPermission(player.userID, "MyPlugin.my_permission") )
    {
    // Do some stuff
    }
