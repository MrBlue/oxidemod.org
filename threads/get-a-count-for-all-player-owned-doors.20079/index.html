    [DOUBLEPOST=1470202804][/DOUBLEPOST]Nevermind got it working. Had to play some trickery.
    Code:
    {% raw %}
            public object doorCount(BasePlayer player)
        {
            var i = 0;
            StabilityEntity[] Door = GameObject.FindObjectsOfType<StabilityEntity>();
            foreach(StabilityEntity ddoor in Door) if(ddoor.OwnerID == player.userID) if(ddoor.OwnerID == player.userID && ddoor.ToString().Contains("door.hinged")) i++;
            return i;
        }
    {% endraw %}
