    this would for example stop the Explode:

    Code:
    {% raw %}
    void OnExplosiveThrown(BasePlayer player, BaseEntity entity)
{
    if (entity != null && entity is TimedExplosive)
    {
        entity.CancelInvoke("Explode");
    }
}
    {% endraw %}
