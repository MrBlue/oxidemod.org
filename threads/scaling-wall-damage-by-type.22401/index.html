    I'm trying to figure out how to return the type of wall when a player hits it and then scale it based on what it returns.

    So I want to do different multipliers for slash based on the wall grade.



    I think i've figured it out

    Code:
    {% raw %}
    private void OnEntityTakeDamage(BaseCombatEntity entity, HitInfo hitInfo)
         {            // if (entity is BuildingBlock)
            //for (var i = 0; i < DamageTypeMax; i++)
            //{
            //    hitInfo.damageTypes.Scale((DamageType) i, _modifiers[i]);
            //}           
            if (entity is BuildingBlock)
            {
                var block = entity as BuildingBlock;
               
                if (block.grade == BuildingGrade.Enum.Wood)
                    hitInfo.damageTypes.Scale(Rust.DamageType.Slash, 2);
                if (block.grade == BuildingGrade.Enum.Stone)
                    hitInfo.damageTypes.Scale(Rust.DamageType.Slash, 20);
               
            }            if (entity is Door)
                hitInfo.damageTypes.Scale(Rust.DamageType.Slash, 5);            else return;
         }
