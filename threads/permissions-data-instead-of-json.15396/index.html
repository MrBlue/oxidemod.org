    grant <group|user> <name|id> *|<permission>
    revoke <group|user> <name|id> *|<permission>
    group <add|set> <name> [title] [rank]
    group <remove> <name>
    group <parent> <name> <parentname>
    usergroup <add|remove> <username> <groupname>
    show <group|user> <name>
    show <groups|perms>
