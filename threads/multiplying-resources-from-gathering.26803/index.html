  2. LaserHydra

    Code:
    {% raw %}
    void OnCollectiblePickup(LootOnPickup node, WorldItemInteractServer player, List<ItemObject> items)
{
    foreach (var item in items)
    {
        item.StackSize *= 5;
        item.InvalidateStack();
    }
}
    {% endraw %}
