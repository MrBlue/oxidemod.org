  6. Calytic Community Admin Community Mod

    Code:
    {% raw %}
    SomeFunc: function(table)
{
    var userfield = table.GetType().GetField("user", rust.PrivateBindingFlag());
    var player = userfield.GetValue(table);
}
    {% endraw %}
