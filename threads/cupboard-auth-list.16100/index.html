  12. k1lly0u

    this is how you could do it in C#, but for LUA you are on your own
    Code:
    {% raw %}
    void GetAuthList(BaseEntity entity)
        {
            if (entity is BuildingPrivlidge)
            {
                List<string> authedPlayers = new List<string>();
                var tc = entity.GetComponent<BuildingPrivlidge>();
                if (tc != null)
                    foreach (var entry in tc.authorizedPlayers)
                        authedPlayers.Add(entry.username);
            }
        }
    {% endraw %}
    entry.userid will give you ulong ID
