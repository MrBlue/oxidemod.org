  4. sami37

    When i did some of my plugin i was working like that
    Code:
    {% raw %}
    private readonly Core.MySql.Libraries.MySql _mySql = Interface.GetMod().GetLibrary<Core.MySql.Libraries.MySql>();
Connection Sql_conn;
    {% endraw %}
    Opening database :
    Code:
    {% raw %}
    Sql_conn = _mySql.OpenDb("host", "port", "db_name", "user", "password", this);
    {% endraw %}
    Select query :
    Code:
    {% raw %}
    var sql = Core.Database.Sql.Builder.Append("SELECT * FROM tablename");
_mySql.Query(sql, Sql_conn, list =>
{
    if (list == null || list.Count == 0)
    {
        //Do your need here
    }
});
    {% endraw %}
    Update query:
    Code:
    {% raw %}
    var updatesql = Core.Database.Sql.Builder.Append("UPDATE tablename SET column = value WHERE column = value;");
_mySql.Update(updatesql, Sql_conn);
    {% endraw %}
    Insert query:
    Code:
    {% raw %}
    var insert = Core.Database.Sql.Builder.Append("INSERT INTO table_name ( `Col1`, `Col2`) VALUES ('Val1', 'Val2');
_mySql.Insert(insert, Sql_conn);
    {% endraw %}
    Close database:
    Code:
    {% raw %}
    _mySql.CloseDb(Sql_conn);
    {% endraw %}
