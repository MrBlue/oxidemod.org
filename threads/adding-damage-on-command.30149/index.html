  2. xMrVizzy

    Code:
    {% raw %}
            [ChatCommand("ad")]
        private void AdCommand(BasePlayer player, string command, string[] args)
        {
            var damage = Convert.ToInt32(args[0]);
           
            if ((player.health - damage) < 1) player.Die();
            else player.health -= damage;
        }
    {% endraw %}
