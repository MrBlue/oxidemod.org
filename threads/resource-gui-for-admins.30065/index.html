            readonly static List<string> objects_your_looking_for = new List<string>() { "item1", "item2" };
        static Dictionary<string, int> list_of_objects = new Dictionary<string, int>();
     
        public static string Repeat(int count, string st)
        {
            for (int i = 0; i < count; i++)
            {
                if (st.ToLower().Contains(objects_your_looking_for[i])){
                    if (!list_of_objects.ContainsKey(objects_your_looking_for[i]))
                    {
                        list_of_objects.Add(objects_your_looking_for[i], 0);
                    }
                    return objects_your_looking_for[i];
                }
            }
            return null;
        }
        void foo()
        {
            foreach (var mach in GameObject.FindObjectsOfType<SimpleBuildingBlock>())
            {
                var name = mach?.name ?? string.Empty;
                string name_found = Repeat(objects_your_looking_for.Count, name);
                if (name_found != null)
                {
                    list_of_objects[name_found] += 1;
                }
            }
        }
        [ChatCommand("checkallNodes")]
        void getAllTheStuffs(BasePlayer player)
        {
            if (player.IsAdmin)
            {
                foo();                foreach(string s in list_of_objects.Keys)
                {
                    player.SendMessage(s + ": " + list_of_objects[s]);
                }
            }
        }
